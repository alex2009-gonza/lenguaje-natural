#chatbot con librerias NLTK
import nltk # importo libreria de
from nltk.tokenize import word_tokenize #importo toquenizador de nltk
import random #importo para saleccionar aleatorias

#descargando recursos para tokenizar (solo en la primer ejecucuion
nltk.download('punkt',quiet=True)# tokemizar punkt

#base de datos de respuestas del chatbot
respuestas={
    "hola":["hola como estaas","que tal","q haces","para q me necesitas","q ps peje"],
    "bye":["hasta luego","me dio gusto verte, adios","largateee jsjs","goodbye"],
    "hombre":["hola soy chato","soy chato","soy tu asistente virtual","soy un chabot me llamo chato"],
    "gracias":["de nada","con gusto","estoy para ayudarte"],
}

def buscar_respuestas(mensaje):
    #tokenizando el mensaje recibido
    mensaje=set(word_tokenize(mensaje.lower()))
    #hola como estas
    coincidencias=mensaje & respuestas.keys()

    if coincidencias:
        palabra_encontrada=coincidencias.pop()
        #saca la palabra encontrada del conjunto de las palabras
        return random.choice(respuestas[palabra_encontrada])

    return "No entiendo tu mensaje,escribelo de nuevo"
    #termina la funsion buscar_respuestas

print("chatbot:hola! escribe 'bye' para salir")

while True:
    mensaje=input("tu: ")
    if mensaje.lower()=="bye":
        print("chatbot:Adios")
        break

        respuestas=buscar_respuestas(mensaje)
        print("chatbot:",respuestas,"\n")
